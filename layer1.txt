    <script>
        const video = document.getElementById('bg-video');
        const overlay = document.getElementById('start-overlay');
        const secretKey = document.getElementById('secret-key');

        // 1. Function to Start the Video
        function startStory() {
            video.play();
            overlay.style.opacity = '0'; // Fade out overlay
            setTimeout(() => {
                overlay.style.display = 'none'; // Remove it completely
            }, 500);
        }

        // 2. Detect when video ends
        video.addEventListener('ended', () => {
            console.log("Story finished. Secret key active.");
            secretKey.style.display = 'block'; // Make the invisible button active
        });

        // 3. Navigation
        function goToLayer2() {
            // alert("Secret Key Found! Going to Layer 2...");
            window.location.href = "layer2.html"; 
        }
    </script>
